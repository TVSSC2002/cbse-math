\section{\textbf{Matrix}}

-3 & 6\\
-2 & 4\\

    \mydet{a^{2} + 1 & ab & ac\\
                ab & b^{2} + 1 & bc\\
		ac & bc & c^{2} + 1} = 1 + a^{2} + b^{2} + c^{2}
\end{align*}
\item If $A = \myvec{
		1 & -1 & 1\\
		2 & -1 & 0\\
		1 & 0 & 0
	}$, find $ A^{2}$ and show that $A^{2}
 = A^{-1}$.	
\item Using matrix method,solve the following system of equations:
	\begin{align*}
		2x - 3y + 5z &= 13\\
		3x + 2y - 4z &= -2\\
		x + y - 2z &= -2
	\end{align*}
